# 3. 无重复字符的最长子串

```js
/**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function (s) {
  const set = new Set();
  const n = s.length;
  if (n < 2) return n;
  let ans = 0;
  for (let left = 0, right = 0; right < n; right++) {
    while (set.has(s[right])) {
      set.delete(s[left++]);
    }
    set.add(s[right]);
    ans = Math.max(ans, right - left + 1);
  }
  return ans;
};
```
