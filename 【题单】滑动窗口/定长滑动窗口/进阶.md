# 2134. 最少交换次数来组合所有的 1 II

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var minSwaps = function (nums) {
  let k = 0;
  for (const num of nums) {
    if (num === 1) {
      k++;
    }
  }
  let count = 0;
  let maxCount = 0;
  nums = [...nums, ...nums];
  const n = nums.length;
  for (let i = 0; i < n; i++) {
    count += nums[i];
    if (i < k - 1) {
      continue;
    }
    maxCount = Math.max(count, maxCount);
    count -= nums[i - k + 1];
  }
  return k - maxCount;
};
```
