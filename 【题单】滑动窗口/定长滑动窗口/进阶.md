# 2134. 最少交换次数来组合所有的 1 II

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var minSwaps = function (nums) {
  let k = 0;
  for (const num of nums) {
    if (num === 1) {
      k++;
    }
  }
  let count = 0;
  let maxCount = 0;
  nums = [...nums, ...nums];
  const n = nums.length;
  for (let i = 0; i < n; i++) {
    count += nums[i];
    if (i < k - 1) {
      continue;
    }
    maxCount = Math.max(count, maxCount);
    count -= nums[i - k + 1];
  }
  return k - maxCount;
};
```

# 567. 字符串的排列

```js
/**
 * @param {string} s1
 * @param {string} s2
 * @return {boolean}
 */
var checkInclusion = function (s1, s2) {
  const k = s1.length;
  const n = s2.length;
  if (k > n) {
    return false;
  }
  const cnt1 = new Array(26).fill(0);
  const cnt2 = new Array(26).fill(0);
  for (let i = 0; i < n; i++) {
    cnt2[s2[i].charCodeAt(0) - "a".charCodeAt(0)]++;
    if (i < k) {
      cnt1[s1[i].charCodeAt(0) - "a".charCodeAt(0)]++;
    }
    if (i < k - 1) {
      continue;
    }
    if (cnt1.join("") === cnt2.join("")) {
      return true;
    }
    cnt2[s2[i - k + 1].charCodeAt(0) - "a".charCodeAt(0)]--;
  }
  return false;
};
```
