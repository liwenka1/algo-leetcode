# 1456. 定长子串中元音的最大数目

```js
/**
 * @param {string} s
 * @param {number} k
 * @return {number}
 */
var maxVowels = function (s, k) {
  const set = new Set(["a", "e", "i", "o", "u"]);
  const n = s.length;
  let ans = 0;
  let temp = ans;
  for (let i = 0; i < n; i++) {
    if (set.has(s[i])) {
      temp++;
    }
    if (i < k - 1) {
      continue;
    }
    ans = Math.max(ans, temp);
    if (set.has(s[i - k + 1])) {
      temp--;
    }
  }
  return ans;
};
```

# 2269. 找到一个数字的 K 美丽值

```js
/**
 * @param {number} num
 * @param {number} k
 * @return {number}
 */
var divisorSubstrings = function (num, k) {
  num = String(num);
  const n = num.length;
  let ans = 0;
  for (let i = k; i <= n; i++) {
    const str = num.slice(i - k, i);
    if (num % str === 0) {
      ans++;
    }
  }
  return ans;
};
```

# 1984. 学生分数的最小差值

```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var minimumDifference = function (nums, k) {
  nums.sort((a, b) => a - b);
  const n = nums.length;
  let ans = Infinity;
  for (let i = 0; i < n - k + 1; i++) {
    ans = Math.min(ans, nums[i + k - 1] - nums[i]);
  }
  return ans;
};
```

# 643. 子数组最大平均数 I

```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var findMaxAverage = function (nums, k) {
  const n = nums.length;
  let sum = 0;
  let ans = -Infinity;
  for (let i = 0; i < n; i++) {
    sum += nums[i];
    if (i < k - 1) {
      continue;
    }
    ans = Math.max(ans, sum / k);
    sum -= nums[i - k + 1];
  }
  return ans;
};
```

# 1343. 大小为 K 且平均值大于等于阈值的子数组数目

```js
/**
 * @param {number[]} arr
 * @param {number} k
 * @param {number} threshold
 * @return {number}
 */
var numOfSubarrays = function (arr, k, threshold) {
    const n = arr.length;
    let sum = 0;
    let ans = 0;
    for (let i = 0; i < n; i++) {
        sum += arr[i];
        if (i < k - 1) {
            continue;
        }
        if (sum / k >= threshold) {
            ans++;
        }
        sum -= arr[i - k + 1];
    }
    return ans;
};
```
