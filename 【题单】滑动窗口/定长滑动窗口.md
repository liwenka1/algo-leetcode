# 1456. 定长子串中元音的最大数目

```js
/**
 * @param {string} s
 * @param {number} k
 * @return {number}
 */
var maxVowels = function (s, k) {
  const set = new Set(["a", "e", "i", "o", "u"]);
  const n = s.length;
  let ans = 0;
  let temp = ans;
  for (let i = 0; i < n; i++) {
    if (set.has(s[i])) {
      temp++;
    }
    if (i < k - 1) {
      continue;
    }
    ans = Math.max(ans, temp);
    if (set.has(s[i - k + 1])) {
      temp--;
    }
  }
  return ans;
};
```
