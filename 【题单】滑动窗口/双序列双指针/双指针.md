# 2540. 最小公共值

```js
/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var getCommon = function (nums1, nums2) {
  const n = nums1.length;
  const m = nums2.length;
  let i = 0;
  let j = 0;
  while (i < n && j < m) {
    if (nums1[i] < nums2[j]) {
      i++;
    } else if (nums1[i] > nums2[j]) {
      j++;
    } else {
      return nums1[i];
    }
  }
  return -1;
};
```
