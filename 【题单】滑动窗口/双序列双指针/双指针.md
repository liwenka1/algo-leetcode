```js
 * @return {number}
 */
var getCommon = function (nums1, nums2) {
  const n = nums1.length;
  const m = nums2.length;
  let i = 0;
  let j = 0;
  while (i < n && j < m) {
    if (nums1[i] < nums2[j]) {
      i++;
    } else if (nums1[i] > nums2[j]) {
      j++;
    } else {
      return nums1[i];
    }
  }
  return -1;
};
```

# 88. 合并两个有序数组

```js
/**
 * @param {number[]} nums1
 * @param {number} m
 * @param {number[]} nums2
 * @param {number} n
 * @return {void} Do not return anything, modify nums1 in-place instead.
 */
var merge = function (nums1, m, nums2, n) {
  let i = m - 1;
  let j = n - 1;
  let p = m + n - 1;
  while (j >= 0) {
    if (i >= 0 && nums1[i] > nums2[j]) {
      nums1[p--] = nums1[i--];
    } else {
      nums1[p--] = nums2[j--];
    }
  }
};
```

```js
 */
var mergeArrays = function (nums1, nums2) {
  const n = nums1.length;
  const m = nums2.length;
  let i = 0;
  let j = 0;
  const ans = [];
  while (i < n && j < m) {
    const [idi, vali] = nums1[i];
    const [idj, valj] = nums2[j];
    if (idi < idj) {
      ans.push(nums1[i]);
      i++;
    } else if (idi > idj) {
      ans.push(nums2[j]);
      j++;
    } else {
      ans.push([idi, vali + valj]);
      i++;
      j++;
    }
  }
  while (i < n) {
    ans.push(nums1[i]);
    i++;
  }
  while (j < m) {
    ans.push(nums2[j]);
    j++;
  }
  return ans;
};
```

# LCP 18. 早餐组合

```js
/**
 * @param {number[]} staple
 * @param {number[]} drinks
 * @param {number} x
 * @return {number}
 */
var breakfastNumber = function (staple, drinks, x) {
  staple.sort((a, b) => a - b);
  drinks.sort((a, b) => a - b);
  const MOD = 1e9 + 7;
  const n = staple.length;
  const m = drinks.length;
  let i = 0;
  let j = m - 1;
  let ans = 0;
  while (i < n && j >= 0) {
    const sum = staple[i] + drinks[j];
    if (sum <= x) {
      ans += j + 1;
      ans %= MOD;
      i++;
    } else {
      j--;
    }
  }
  return ans;
};
```

# 1385. 两个数组间的距离值

```js
/**
 * @param {number[]} arr1
 * @param {number[]} arr2
 * @param {number} d
 * @return {number}
 */
const binarySearch = (arr, target) => {
  let low = 0;
  let high = arr.length - 1;
  if (arr[high] < target) {
    return high + 1;
  }
  while (low < high) {
    const mid = Math.floor((high - low) / 2) + low;
    if (arr[mid] < target) {
      low = mid + 1;
    } else {
      high = mid;
    }
  }
  return low;
};
var findTheDistanceValue = function (arr1, arr2, d) {
  arr2.sort((a, b) => a - b);
  const n = arr2.length;
  let ans = 0;
  for (const x of arr1) {
    const p = binarySearch(arr2, x);
    let ok = true;
    if (p < n) {
      ok = ok && arr2[p] - x > d;
    }
    if (p > 0) {
      ok = ok && x - arr2[p - 1] > d;
    }
    ans += ok ? 1 : 0;
  }
  return ans;
};
```

# 1855. 下标对中的最大距离

```js
/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var maxDistance = function (nums1, nums2) {
  const n = nums1.length;
  const m = nums2.length;
  let ans = 0;
  for (let j = 0, i = 0; j < m; j++) {
    while (i < n && nums1[i] > nums2[j]) {
      i++;
    }
    if (i < n) {
      ans = Math.max(ans, j - i);
    }
  }
  return ans;
};
```

```js
  const m = typed.length;
  let i = 0;
  let j = 0;
```

```js
  const n = s.length;
  const m = word.length;

  let j = 0;

      return false;
    }

    }
```

```js
var canChange = function (start, target) {
  const n = start.length;
  let i = 0;
  let j = 0;
  while (i < n || j < n) {
    while (i < n && start[i] === "_") {
      i++;
    }
    while (j < n && target[j] === "_") {
      j++;
    }
    if (i === n && j === n) {
      break;
    } else if (i === n || j === n) {
      return false;
    }
    const str = start[i];
    if (target[j] !== str) {
      return false;
    }
    if (str === "L" && i < j) {
      return false;
    }
    if (str === "R" && i > j) {
      return false;
    }
    i++;
    j++;
  }
  return true;
};
```
