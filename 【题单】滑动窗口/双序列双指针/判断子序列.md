# 392. 判断子序列

```js
/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isSubsequence = function (s, t) {
  const n = s.length;
  const m = t.length;
  let i = 0;
  let j = 0;
  while (i < n && j < m) {
    if (s[i] === t[j]) {
      i++;
      j++;
    }
    while (j < m && s[i] !== t[j]) {
      j++;
    }
  }
  return i === n;
};
```
