# 392. 判断子序列

```js
/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isSubsequence = function (s, t) {
  const n = s.length;
  const m = t.length;
  let i = 0;
  let j = 0;
  while (i < n && j < m) {
    if (s[i] === t[j]) {
      i++;
      j++;
    }
    while (j < m && s[i] !== t[j]) {
      j++;
    }
  }
  return i === n;
};
```

```js
/**
 * @param {string} s
 * @param {string} t
 * @return {number}
 */
var appendCharacters = function (s, t) {
  const n = s.length;
  const m = t.length;
  let i = 0;
  let j = 0;
  while (i < n && j < m) {
    if (s[i] === t[j]) {
      i++;
      j++;
    }
    while (i < n && s[i] !== t[j]) {
      i++;
    }
  }
  return m - j;
};
```

# 2825. 循环增长使字符串子序列等于另一个字符串

```js
/**
 * @param {string} str1
 * @param {string} str2
 * @return {boolean}
 */
var canMakeSubsequence = function (str1, str2) {
  const n = str1.length;
  const m = str2.length;
  if (n < m) {
    return false;
  }
  let j = 0;
  for (let i = 0; i < n; i++) {
    const s =
      str1[i] === "z" ? "a" : String.fromCharCode(str1[i].charCodeAt(0) + 1);
    if (s === str2[j] || str1[i] === str2[j]) {
      j++;
      if (j === m) {
        return true;
      }
    }
  }
  return false;
};
```
