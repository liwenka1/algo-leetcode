```js
/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isSubsequence = function (s, t) {
  const n = s.length;
  const m = t.length;
  let i = 0;
  let j = 0;
  while (i < n && j < m) {
    if (s[i] === t[j]) {
      i++;
      j++;
    }
    while (j < m && s[i] !== t[j]) {
      j++;
    }
  }
  return i === n;
};
```

# 2825. 循环增长使字符串子序列等于另一个字符串

```js
/**
 * @param {string} str1
 * @param {string} str2
 * @return {boolean}
 */
var canMakeSubsequence = function (str1, str2) {
  const n = str1.length;
  const m = str2.length;
  if (n < m) {
    return false;
  }
  let j = 0;
  for (let i = 0; i < n; i++) {
    const s =
      str1[i] === "z" ? "a" : String.fromCharCode(str1[i].charCodeAt(0) + 1);
    if (s === str2[j] || str1[i] === str2[j]) {
      j++;
      if (j === m) {
        return true;
      }
    }
  }
  return false;
};
```

# 1023. 驼峰式匹配

```js
/**
 * @param {string[]} queries
 * @param {string} pattern
 * @return {boolean[]}
 */
var camelMatch = function (queries, pattern) {
  const n = queries.length;
  const ans = [];
  for (let i = 0; i < n; i++) {
    ans[i] = true;
    const query = queries[i];
    const m = query.length;
    let k = 0;
    for (let j = 0; j < m; j++) {
      const s = query[j];
      if (k < pattern.length && s === pattern[k]) {
        k++;
      } else if (s.toUpperCase() === s) {
        ans[i] = false;
        break;
      }
    }
    if (k < pattern.length) {
      ans[i] = false;
    }
  }
  return ans;
};
```

# 3132. 找出与数组相加的整数 II

```js
/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var minimumAddedInteger = function (nums1, nums2) {
  nums1.sort((a, b) => a - b);
  nums2.sort((a, b) => a - b);
  const n = nums1.length;
  const m = nums2.length;
  for (let i = 2; i > 0; i--) {
    const x = nums2[0] - nums1[i];
    let k = 0;
    for (let j = i; j < n; j++) {
      if (k < m && nums1[j] + x === nums2[k]) {
        k++;
      }
    }
    if (k === m) {
      return x;
    }
  }
  return nums2[0] - nums1[0];
};
```
