# 125. 验证回文串

```js
/**
 * @param {string} s
 * @return {boolean}
 */
var isPalindrome = function (s) {
  const str = s.replace(/[^a-zA-Z0-9]/g, "").toLowerCase();
  return str === str.split("").reverse().join("");
};
```

# 1750. 删除字符串两端相同字符后的最短长度

```js
/**
 * @param {string} s
 * @return {number}
 */
var minimumLength = function (s) {
  const n = s.length;
  let left = 0;
  let right = n - 1;
  while (left < right && s[left] === s[right]) {
    const str = s[left];
    while (left <= right && s[left] === str) {
      left++;
    }
    while (left <= right && str === s[right]) {
      right--;
    }
  }
  return right - left + 1;
};
```

# 2105. 给植物浇水 II

```js
/**
 * @param {number[]} plants
 * @param {number} capacityA
 * @param {number} capacityB
 * @return {number}
 */
var minimumRefill = function (plants, capacityA, capacityB) {
  const n = plants.length;
  let sA = capacityA;
  let sB = capacityB;
  let left = 0;
  let right = n - 1;
  let ans = 0;
  while (left < right) {
    if (capacityA < plants[left]) {
      capacityA = sA;
      ans++;
    }
    capacityA -= plants[left++];
    if (capacityB < plants[right]) {
      capacityB = sB;
      ans++;
    }
    capacityB -= plants[right--];
  }
  if (left === right) {
    if (capacityA < plants[left] && capacityA >= capacityB) {
      ans++;
    }
    if (capacityB < plants[right] && capacityA < capacityB) {
      ans++;
    }
  }
  return ans;
};
```

# 977. 有序数组的平方

```js
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortedSquares = function (nums) {
  const n = nums.length;
  let left = 0;
  let right = n - 1;
  let ans = new Array(n);
  for (let i = n - 1; i >= 0; i--) {
    const x = nums[left] * nums[left];
    const y = nums[right] * nums[right];
    if (x > y) {
      ans[i] = x;
      left++;
    } else {
      ans[i] = y;
      right--;
    }
  }
  return ans;
};
```

# 658. 找到 K 个最接近的元素

```js
/**
 * @param {number[]} arr
 * @param {number} k
 * @param {number} x
 * @return {number[]}
 */
var findClosestElements = function (arr, k, x) {
  const n = arr.length;
  let left = 0;
  let right = n - 1;
  k = n - k;
  while (k--) {
    if (Math.abs(arr[left] - x) <= Math.abs(arr[right] - x)) {
      right--;
    } else {
      left++;
    }
  }
  return arr.slice(left, right + 1);
};
```

# 1471. 数组中的 k 个最强值

```js
/**
 * @param {number[]} arr
 * @param {number} k
 * @return {number[]}
 */
var getStrongest = function (arr, k) {
  arr.sort((a, b) => a - b);
  const n = arr.length;
  const m = Math.floor((n - 1) / 2);
  const mid = arr[m];
  let left = 0;
  let right = n - 1;
  const ans = [];
  while (k--) {
    if (Math.abs(arr[left] - mid) <= Math.abs(arr[right] - mid)) {
      ans.push(arr[right--]);
    } else {
      ans.push(arr[left++]);
    }
  }
  return ans;
};
```
