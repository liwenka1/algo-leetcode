```js
/**
 * @param {string} s
 * @return {number}
 */
var minimumLength = function (s) {
  const n = s.length;
  let left = 0;
  let right = n - 1;
  while (left < right && s[left] === s[right]) {
    const str = s[left];
    while (left <= right && s[left] === str) {
      left++;
    }
    while (left <= right && str === s[right]) {
      right--;
    }
  }
  return right - left + 1;
};
```

```js
/**
 * @param {number[]} plants
 * @param {number} capacityA
 * @param {number} capacityB
 * @return {number}
 */
var minimumRefill = function (plants, capacityA, capacityB) {
  const n = plants.length;
  let sA = capacityA;
  let sB = capacityB;
  let left = 0;
  let right = n - 1;
  let ans = 0;
  while (left < right) {
    if (capacityA < plants[left]) {
      capacityA = sA;
      ans++;
    }
    capacityA -= plants[left++];
    if (capacityB < plants[right]) {
      capacityB = sB;
      ans++;
    }
    capacityB -= plants[right--];
  }
  if (left === right) {
    if (capacityA < plants[left] && capacityA >= capacityB) {
      ans++;
    }
    if (capacityB < plants[right] && capacityA < capacityB) {
      ans++;
    }
  }
  return ans;
};
```

```js
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortedSquares = function (nums) {
  const n = nums.length;
  let left = 0;
  let right = n - 1;
  let ans = new Array(n);
  for (let i = n - 1; i >= 0; i--) {
    const x = nums[left] * nums[left];
    const y = nums[right] * nums[right];
    if (x > y) {
      ans[i] = x;
      left++;
    } else {
      ans[i] = y;
      right--;
    }
  }
  return ans;
};
```

# 658. 找到 K 个最接近的元素

```js
/**
 * @param {number[]} arr
 * @param {number} k
 * @param {number} x
 * @return {number[]}
 */
var findClosestElements = function (arr, k, x) {
  const n = arr.length;
  let left = 0;
  let right = n - 1;
  k = n - k;
  while (k--) {
    if (Math.abs(arr[left] - x) <= Math.abs(arr[right] - x)) {
      right--;
    } else {
      left++;
    }
  }
  return arr.slice(left, right + 1);
};
```

```js
        }
```
```js
        left++;
```

```js
          right--;
        }
```

```js
          left++;
        }

      }
```
```js
    }
```

```js
var numTriplets = function (nums1, nums2) {

    for (let j = 0; j < n - 1; j++) {
      for (let k = j + 1; k < n; k++) {
        const num = n2[j] * n2[k];
        map.set(num, (map.get(num) || 0) + 1);
```

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var threeSumMulti = function (nums, target) {
  const MOD = 1000000007;
  nums.sort((a, b) => a - b);
  const n = nums.length;
  let ans = 0;
  for (let i = 0; i < n - 2; i++) {
    if (nums[i] + nums[n - 1] + nums[n - 2] < target) {
      continue;
    }
    if (nums[i] + nums[i + 1] + nums[i + 2] > target) {
      break;
    }
    let left = i + 1;
    let right = n - 1;
    while (left < right) {
      const sum = nums[i] + nums[left] + nums[right];
      if (sum > target) {
        right--;
      }
      if (sum < target) {
        left++;
      }
      if (sum === target) {
        if (nums[left] === nums[right]) {
          ans += ((right - left + 1) * (right - left)) / 2;
          ans %= MOD;
          break;
        }
        let leftcnt = 1;
        let rightcnt = 1;
        while (nums[left] === nums[left + 1]) {
          left++;
          leftcnt++;
        }
        while (nums[right] === nums[right - 1]) {
          right--;
          rightcnt++;
        }
        left++;
        right--;
        ans += leftcnt * rightcnt;
        ans %= MOD;
      }
    }
  }
  return ans;
};
```

```js
var bagOfTokensScore = function (tokens, power) {
  tokens.sort((a, b) => a - b);
  const n = tokens.length;
  let left = 0;
  let right = n - 1;
  let ans = 0;
  while (left <= right) {
    if (tokens[left] <= power) {
      ans++;
      power -= tokens[left++];
    } else if (ans >= 1 && left < right) {
      ans--;
      power += tokens[right--];
    } else {

    }
  }

};
```
