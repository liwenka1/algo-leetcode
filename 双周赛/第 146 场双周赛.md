# 3393. 统计异或值为给定值的路径数目

```js
/**
 * @param {number[][]} grid
 * @param {number} k
 * @return {number}
 */
var countPathsWithXorValue = function (grid, k) {
  const n = grid.length;
  const m = grid[0].length;
  const LIM = 16;
  const MOD = 1e9 + 7;

  const dp = Array.from({ length: n }, () =>
    Array.from({ length: m }, () => new Array(LIM).fill(0))
  );

  dp[0][0][grid[0][0]] = 1;

  for (let i = 0; i < n; i++) {
    for (let j = 0; j < m; j++) {
      if (i + j > 0) {
        for (let k = 0; k < LIM; k++) {
          if (i > 0) {
            dp[i][j][k] = (dp[i][j][k] + dp[i - 1][j][k ^ grid[i][j]]) % MOD;
          }

          if (j > 0) {
            dp[i][j][k] = (dp[i][j][k] + dp[i][j - 1][k ^ grid[i][j]]) % MOD;
          }
        }
      }
    }
  }

  return dp[n - 1][m - 1][k];
};
```
