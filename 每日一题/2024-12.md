# 51. N 皇后

```js
/**
 * @param {number} n
 * @return {string[][]}
 */
var solveNQueens = function (n) {
  const path = Array.from({ length: n }, () => new Array(n).fill("."));
  const cols = new Set();
  const diag1 = new Set();
  const diag2 = new Set();
  const ans = [];
  const dfs = (row) => {
    if (row === n) {
      ans.push(path.map((rows) => rows.join("")));
      return;
    }

    for (let col = 0; col < n; col++) {
      if (cols.has(col) || diag1.has(row + col) || diag2.has(row - col)) {
        continue;
      }

      cols.add(col);
      diag1.add(row + col);
      diag2.add(row - col);
      path[row][col] = "Q";

      dfs(row + 1);

      cols.delete(col);
      diag1.delete(row + col);
      diag2.delete(row - col);
      path[row][col] = ".";
    }
  };
  dfs(0);
  return ans;
};
```

```js
var minMovesToCaptureTheQueen = function (a, b, c, d, e, f) {
  const inBetween = (l, r, m) => {
    return Math.min(l, r) < m && m < Math.max(l, r);
  };
  if (

    return 1;

    (c - d === e - f && (a - b !== e - f || !inBetween(c, e, a)))

  }
```

```js
 */
var numRookCaptures = function (board) {
  const size = 8;
  let x0 = 0;
  let y0 = 0;
  for (let i = 0; i < size; i++) {
    for (let j = 0; j < size; j++) {
      if (board[i][j] === "R") {
        x0 = i;
        y0 = j;
      }
    }
  }
  const distance = [
    [0, -1],
    [0, 1],
    [-1, 0],
    [1, 0],
  ];
  let ans = 0;
  for (const [x, y] of distance) {
    let x1 = x + x0;
    let y1 = y + y0;
    while (
      x1 >= 0 &&
      x1 < size &&
      y1 >= 0 &&
      y1 < size &&
      board[x1][y1] === "."
    ) {
      x1 += x;
      y1 += y;
    }
    if (x1 >= 0 && x1 < size && y1 >= 0 && y1 < size && board[x1][y1] === "p") {
      ans++;
    }
  }
  return ans;
};
```

```js
 */
var semiOrderedPermutation = function (nums) {
  const n = nums.length;
  if (nums[0] === 1 && nums[n - 1] === n) {
    return 0;
  }
  let firstindex = nums.indexOf(1);
  let finalindex = nums.indexOf(n);
  return finalindex > firstindex
    ? n - 1 - finalindex + firstindex
    : n - 1 - finalindex + firstindex - 1;
};
```

```js
var getKth = function (lo, hi, k) {
  const getCnt = (num) => {
    let res = 0;
    while (num !== 1) {
      if (num % 2 === 0) {
        num = num / 2;
      } else {
        num = 3 * num + 1;
      }
      res++;
    }
    return res;
  };
  const ans = [];
  for (let i = lo; i <= hi; i++) {
    ans.push([i, getCnt(i)]);
  }
  ans.sort((a, b) => {
    if (a[1] !== b[1]) {
      return a[1] - b[1];
    } else {
      return a[0] - b[0];
    }
  });
  return ans[k - 1][0];
};
```

```js
var eatenApples = function (apples, days) {

      pq.dequeue();

    }
  }
  return ans;
```

```js
var rankTeams = function (votes) {
  const map = {};
  for (const vote of votes) {
    const n = vote.length;
    for (let i = 0; i < n; i++) {
      const s = vote[i];
      if (!map[s]) {
        map[s] = new Array(n).fill(0);
      }
      map[s][i]++;
    }
  }

    .sort()
    .sort((a, b) => {
      const n = map[a].length;

          return map[b][i] - map[a][i];
        }
      }
    })
    .join("");
};
```

# 1367. 二叉树中的链表

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {ListNode} head
 * @param {TreeNode} root
 * @return {boolean}
 */
var isSubPath = function (head, root) {
  const dfs = (s, t) => {
    if (s === null) {
      return true;
    }

    if (t === null) {
      return false;
    }

    if (s.val === t.val) {
      if (dfs(s.next, t.left) || dfs(s.next, t.right)) {
        return true;
      }
    }

    if (s === head) {
      return dfs(s, t.left) || dfs(s, t.right);
    }

    return false;
  };

  return dfs(head, root);
};
```
