# 51. N 皇后

```js
/**
 * @param {number} n
 * @return {string[][]}
 */
var solveNQueens = function (n) {
  const path = Array.from({ length: n }, () => new Array(n).fill("."));
  const cols = new Set();
  const diag1 = new Set();
  const diag2 = new Set();
  const ans = [];
  const dfs = (row) => {
    if (row === n) {
      ans.push(path.map((rows) => rows.join("")));
      return;
    }

    for (let col = 0; col < n; col++) {
      if (cols.has(col) || diag1.has(row + col) || diag2.has(row - col)) {
        continue;
      }

      cols.add(col);
      diag1.add(row + col);
      diag2.add(row - col);
      path[row][col] = "Q";

      dfs(row + 1);

      cols.delete(col);
      diag1.delete(row + col);
      diag2.delete(row - col);
      path[row][col] = ".";
    }
  };
  dfs(0);
  return ans;
};
```

```js
var totalNQueens = function (n) {
  const path = Array.from({ length: n }, () => new Array(n).fill("."));
  const cols = new Set();
  const diag1 = new Set();
  const diag2 = new Set();
  let ans = 0;
  const dfs = (row) => {
    if (row === n) {
      ans++;
      return;
    }

    for (let col = 0; col < n; col++) {
      if (path[row][col] !== ".") {
        continue;
      }
      if (cols.has(col) || diag1.has(row + col) || diag2.has(row - col)) {
        continue;
      }

      path[row][col] = "Q";
      cols.add(col);
      diag1.add(row + col);
      diag2.add(row - col);

      dfs(row + 1);

      path[row][col] = ".";
      cols.delete(col);
      diag1.delete(row + col);
      diag2.delete(row - col);
    }
  };
  dfs(0);
  return ans;
};
```

```js
/**

var checkTwoChessboards = function (coordinate1, coordinate2) {
  const sum1 = coordinate1.charCodeAt(0) + parseInt(coordinate1[1]);
```

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number[]} nums
 * @return {number}
 */
var numComponents = function (head, nums) {
  const numSet = new Set(nums);
  let inSet = false;
  let ans = 0;
  while (head) {
    if (numSet.has(head.val)) {
      if (!inSet) {
        inSet = true;
        ans++;
      }
    } else {
      inSet = false;
    }
    head = head.next;
  }
  return ans;
};
```

# 2466. 统计构造好字符串的方案数

```js
/**
 * @param {number} a
 * @param {number} b
 * @param {number} c
 * @param {number} d
 * @param {number} e
 * @param {number} f
 * @return {number}
 */
var minMovesToCaptureTheQueen = function (a, b, c, d, e, f) {
  const inBetween = (l, r, m) => {
    return Math.min(l, r) < m && m < Math.max(l, r);
  };
  if (
    (a === e && (c !== e || !inBetween(b, f, d))) ||
    (b === f && (d !== f || !inBetween(a, e, c)))
  ) {
    return 1;
  }
  if (
    (c + d === e + f && (a + b !== e + f || !inBetween(c, e, a))) ||
    (c - d === e - f && (a - b !== e - f || !inBetween(c, e, a)))
  ) {
    return 1;
  }
  return 2;
};
```

# 999. 可以被一步捕获的棋子数

```js
/**
 * @param {character[][]} board
 * @return {number}
 */
var numRookCaptures = function (board) {
  const size = 8;
  let x0 = 0;
  let y0 = 0;
  for (let i = 0; i < size; i++) {
    for (let j = 0; j < size; j++) {
      if (board[i][j] === "R") {
        x0 = i;
        y0 = j;
      }
    }
  }
  const distance = [
    [0, -1],
    [0, 1],
    [-1, 0],
    [1, 0],
  ];
  let ans = 0;
  for (const [x, y] of distance) {
    let x1 = x + x0;
    let y1 = y + y0;
    while (
      x1 >= 0 &&
      x1 < size &&
      y1 >= 0 &&
      y1 < size &&
      board[x1][y1] === "."
    ) {
      x1 += x;
      y1 += y;
    }
    if (x1 >= 0 && x1 < size && y1 >= 0 && y1 < size && board[x1][y1] === "p") {
      ans++;
    }
  }
  return ans;
};
```

```js
/**

 */
var squareIsWhite = function (coordinates) {
  const x = coordinates.charCodeAt(0);
  const y = Number(coordinates[1]);
  return (x + y) % 2 === 1;
```
