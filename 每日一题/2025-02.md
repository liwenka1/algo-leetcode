/**
 * @param {number[]} nums
 * @param {number} target
 * @return {boolean}
 */
var search = function (nums, target) {
  const check = (mid, right) => {
    const x = nums[mid];
    if (x > nums[right]) {
      return target > nums[right] && x >= target;
    }
    return target > nums[right] || x >= target;
  };
  let left = 0;
  let right = nums.length - 1;
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (nums[mid] === target) {
      return true;
    }
    if (nums[mid] === nums[right]) {
      right--;
    } else if (check(mid, right)) {
      right = mid - 1;
    } else {
      left = mid + 1;
    }
  }
  return false;
};

# 598. 区间加法 II

```js
/**
 * @param {number} m
 * @param {number} n
 * @param {number[][]} ops
 * @return {number}
 */
var maxCount = function (m, n, ops) {
  let minA = m;
  let minB = n;
  for (const [a, b] of ops) {
    minA = Math.min(minA, a);
    minB = Math.min(minB, b);
  }
  return minA * minB;
};

/**
 * @param {string} s
 * @return {boolean}
 */
var validPalindrome = function (s) {
  const isPalindrome = (left, right) => {
    while (left <= right) {
      if (s[left] !== s[right]) {
        return false;
      }
      left++;
      right--;
    }
    return true;
  };
  const n = s.length;
  let left = 0;
  let right = n - 1;
  let cnt = 0;
  while (left <= right) {
    if (s[left] !== s[right]) {
      return isPalindrome(left + 1, right) || isPalindrome(left, right - 1);
    }
    left++;
    right--;
  }
  return cnt <= 1;
};

/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortArrayByParityII = function (nums) {
  const n = nums.length;
  for (let i = 0, j = 1; i < n; i += 2) {
    if (nums[i] % 2 !== 0) {
      while (nums[j] % 2 !== 0) {
        j += 2;
      }
      [nums[i], nums[j]] = [nums[j], nums[i]];
    }
  }
  return nums;
};

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var subsetsWithDup = function (nums) {
  nums.sort((a, b) => a - b);
  const n = nums.length;
  const path = [];
  const ans = [];
  const dfs = (start) => {
    ans.push([...path]);
    if (start === n) {
      return;
    }
    for (let i = start; i < n; i++) {
      if (i > start && nums[i] === nums[i - 1]) {
        continue;
      }
      path.push(nums[i]);
      dfs(i + 1);
      path.pop();
    }
  };
  dfs(0);
  return ans;
};

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permuteUnique = function (nums) {
  nums.sort((a, b) => a - b);
  const n = nums.length;
  const used = new Array(n).fill(false);
  const path = new Array(n);
  const ans = [];
  const dfs = (start) => {
    if (start === n) {
      ans.push([...path]);
      return;
    }
    for (let i = 0; i < n; i++) {
      if (used[i] || (i > 0 && nums[i] === nums[i - 1] && !used[i - 1])) {
        continue;
      }
      path[start] = nums[i];
      used[i] = true;
      dfs(start + 1);
      used[i] = false;
    }
  };
  dfs(0);
  return ans;
};

/**
 * @param {number} n
 * @return {number[][]}
 */
var generateMatrix = function (n) {
  const DIRS = [
    [0, 1],
    [1, 0],
    [0, -1],
    [-1, 0],
  ];
  const ans = Array.from({ length: n }, () => new Array(n).fill(0));
  let i = 0;
  let j = 0;
  let di = 0;
  for (let val = 1; val <= n * n; val++) {
    ans[i][j] = val;
    const x = i + DIRS[di][0];
    const y = j + DIRS[di][1];
    if (x < 0 || x >= n || y < 0 || y >= n || ans[x][y] !== 0) {
      di = (di + 1) % 4;
    }
    i += DIRS[di][0];
    j += DIRS[di][1];
  }
  return ans;
};

/**
 * @param {number[][]} obstacleGrid
 * @return {number}
 */
var uniquePathsWithObstacles = function (obstacleGrid) {
  const n = obstacleGrid.length;
  const m = obstacleGrid[0].length;
  const dp = Array.from({ length: n }, () => new Array(m).fill(0));
  for (let i = 0; i < n && obstacleGrid[i][0] === 0; i++) {
    dp[i][0] = 1;
  }
  for (let j = 0; j < m && obstacleGrid[0][j] === 0; j++) {
    dp[0][j] = 1;
  }
  for (let i = 1; i < n; i++) {
    for (let j = 1; j < m; j++) {
      if (obstacleGrid[i][j] === 0) {
        dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
      }
    }
  }
  return dp[n - 1][m - 1];
};
```

# 80. 删除有序数组中的重复项 II

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var removeDuplicates = function (nums) {
  const n = nums.length;
  if (n === 1) {
    return n;
  }
  let stackSize = 2;
  for (let i = 2; i < n; i++) {
    if (nums[i] !== nums[stackSize - 2]) {
      nums[stackSize++] = nums[i];
    }
  }
  return stackSize;
};
```

/**
 * @param {number[]} nums
 * @param {number} maxOperations
 * @return {number}
 */
var minimumSize = function (nums, maxOperations) {
  const check = (mid) => {
    let cnt = 0;
    for (const x of nums) {
      cnt += Math.ceil(x / mid) - 1;
    }
    return cnt <= maxOperations;
  };
  let left = 0;
  let right = Math.max(...nums);
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (check(mid)) {
      right = mid - 1;
    } else {
      left = mid + 1;
    }
  }
  return left;
};
