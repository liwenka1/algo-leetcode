# 81. 搜索旋转排序数组 II

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {boolean}
 */
var search = function (nums, target) {
  const check = (mid, right) => {
    const x = nums[mid];
    if (x > nums[right]) {
      return target > nums[right] && x >= target;
    }
    return target > nums[right] || x >= target;
  };
  let left = 0;
  let right = nums.length - 1;
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (nums[mid] === target) {
      return true;
    }
    if (nums[mid] === nums[right]) {
      right--;
    } else if (check(mid, right)) {
      right = mid - 1;
    } else {
      left = mid + 1;
    }
  }
  return false;
};
```

# 598. 区间加法 II

```js
/**
 * @param {number} m
 * @param {number} n
 * @param {number[][]} ops
 * @return {number}
 */
var maxCount = function (m, n, ops) {
  let minA = m;
  let minB = n;
  for (const [a, b] of ops) {
    minA = Math.min(minA, a);
    minB = Math.min(minB, b);
  }
  return minA * minB;
};
```

# 680. 验证回文串 II

```js
/**
 * @param {string} s
 * @return {boolean}
 */
var validPalindrome = function (s) {
  const isPalindrome = (left, right) => {
    while (left <= right) {
      if (s[left] !== s[right]) {
        return false;
      }
      left++;
      right--;
    }
    return true;
  };
  const n = s.length;
  let left = 0;
  let right = n - 1;
  let cnt = 0;
  while (left <= right) {
    if (s[left] !== s[right]) {
      return isPalindrome(left + 1, right) || isPalindrome(left, right - 1);
    }
    left++;
    right--;
  }
  return cnt <= 1;
};
```

# 922. 按奇偶排序数组 II

```js
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortArrayByParityII = function (nums) {
  const n = nums.length;
  for (let i = 0, j = 1; i < n; i += 2) {
    if (nums[i] % 2 !== 0) {
      while (nums[j] % 2 !== 0) {
        j += 2;
      }
      [nums[i], nums[j]] = [nums[j], nums[i]];
    }
  }
  return nums;
};
```
