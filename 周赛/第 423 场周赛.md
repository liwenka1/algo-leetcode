# 3350. 检测相邻递增子数组 II

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxIncreasingSubarrays = function (nums) {
  const n = nums.length;
  let pre_cnt = 0;
  let cnt = 0;
  let ans = 0;
  for (let i = 0; i < n; i++) {
    cnt++;
    if (i >= n - 1 || nums[i] >= nums[i + 1]) {
      ans = Math.max(ans, Math.floor(cnt / 2), Math.min(cnt, pre_cnt));
      pre_cnt = cnt;
      cnt = 0;
    }
  }
  return ans;
};
```
