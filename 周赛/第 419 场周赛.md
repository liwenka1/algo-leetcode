# 3318. 计算子数组的 x-sum I

```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @param {number} x
 * @return {number[]}
 */
var findXSum = function (nums, k, x) {
  const n = nums.length;
  if (n < x) {
    return [nums.reduce((a, b) => a + b)];
  }
  const ans = [];
  for (let i = 0; i <= n - k; i++) {
    const arr = nums.slice(i, i + k);
    const map = new Map();
    for (const v of arr) {
      map.set(v, (map.get(v) || 0) + 1);
    }
    const mapList = [...map].sort((a, b) => {
      if (a[1] !== b[1]) {
        return b[1] - a[1];
      }
      return b[0] - a[0];
    });
    let cnt = 0;
    const m = mapList.length < x ? mapList.length : x;
    for (let j = 0; j < m; j++) {
      cnt += mapList[j][0] * mapList[j][1];
    }
    ans.push(cnt);
  }
  return ans;
};
```
