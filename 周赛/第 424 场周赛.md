# 3354. 使数组元素等于零

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var countValidSelections = function (nums) {
  let suf = nums.reduce((a, b) => a + b);
  let pre = 0;
  let ans = 0;
  for (const num of nums) {
    if (num) {
      suf -= num;
      pre += num;
    } else if (suf === pre) {
      ans += 2;
    } else if (Math.abs(suf - pre) === 1) {
      ans++;
    }
  }
  return ans;
};
```
