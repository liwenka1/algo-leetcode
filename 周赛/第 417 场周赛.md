# 3304. 找出第 K 个字符 I

```js
/**
 * @param {number} k
 * @return {character}
 */
var kthCharacter = function (k) {
  let word = "a";
  while (word.length < k) {
    const n = word.length;
    let str = "";
    for (let j = 0; j < n; j++) {
      const s = word[j];
      if (s === "z") {
        str += "a";
      } else {
        str += String.fromCharCode(s.charCodeAt(0) + 1);
      }
    }
    word += str;
  }
  return word[k - 1];
};
```
