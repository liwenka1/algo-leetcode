# 3379. 转换数组

```js
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var constructTransformedArray = function (nums) {
  const n = nums.length;
  const ans = new Array(n).fill(0);
  for (let i = 0; i < n; i++) {
    if (nums[i] > 0) {
      ans[i] = nums[(i + nums[i]) % n];
    } else if (nums[i] < 0) {
      ans[i] = nums[(i - Math.abs(nums[i] % n) + n) % n];
    } else {
      ans[i] = nums[i];
    }
  }
  return ans;
};
```
