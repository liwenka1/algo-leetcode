# 3396. 使数组元素互不相同所需的最少操作次数

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var minimumOperations = function (nums) {
  const vis = new Set();
  const n = nums.length;
  for (let i = n - 1; i >= 0; i--) {
    const num = nums[i];
    if (vis.has(num)) {
      return Math.floor(i / 3) + 1;
    }
    vis.add(num);
  }
  return 0;
};
```
