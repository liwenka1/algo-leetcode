# 3386. 按下时间最长的按钮

```js
/**
 * @param {number[][]} events
 * @return {number}
 */
var buttonWithLongestTime = function (events) {
  const n = events.length;
  let [index, time] = events[0];
  for (let i = 1; i < n; i++) {
    const d = events[i][1] - events[i - 1][1];
    if (d > time || (d === time && events[i][0] < index)) {
      [index, time] = [events[i][0], d];
    }
  }
  return index;
};
```

# 3388. 统计数组中的美丽分割

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var beautifulSplits = function (nums) {
  const n = nums.length;
  // 预处理 dp[i][j] 表示以 nums[i] 和 nums[j] 为开头的最长公共前缀
  const dp = Array.from({ length: n + 1 }, () => new Array(n + 1).fill(0));
  for (let i = n - 1; i >= 0; i--) {
    for (let j = n - 1; j > i; j--) {
      if (nums[i] === nums[j]) {
        dp[i][j] = dp[i + 1][j + 1] + 1;
      }
    }
  }
  let ans = 0;
  for (let i = 0; i < n; i++) {
    for (let j = i + 1; j < n - 1; j++) {
      const len1 = i + 1;
      const len2 = j - i;
      const len3 = n - j - 1;
      if (
        (len1 <= len2 && dp[0][i + 1] >= len1) ||
        (len2 <= len3 && dp[i + 1][j + 1] >= len2)
      ) {
        ans++;
      }
    }
  }
  return ans;
};
```
