# 3403. 从盒子中找出字典序最大的字符串 I

```js
/**
 * @param {string} word
 * @param {number} numFriends
 * @return {string}
 */
var answerString = function (word, numFriends) {
  if (numFriends === 1) {
    return word;
  }
  const n = word.length;
  let ans = "";
  for (let i = 0; i < n; i++) {
    const str = word.substring(i, Math.min(i + n - numFriends + 1, n));
    if (str.localeCompare(ans) > 0) {
      ans = str;
    }
  }
  return ans;
};
```
