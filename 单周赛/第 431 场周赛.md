# 3412. 计算字符串的镜像分数

```js
/**
 * @param {string} s
 * @return {number}
 */
var calculateScore = function (s) {
  const path = Array.from({ length: 26 }, () => new Array());
  const n = s.length;
  let ans = 0;
  for (let i = 0; i < n; i++) {
    const index = s.charCodeAt(i) - "a".charCodeAt(0);
    const j = 25 - index;
    if (path[j].length > 0) {
      ans += i - path[j].pop();
    } else {
      path[index].push(i);
    }
  }
  return ans;
};
```
