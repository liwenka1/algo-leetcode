# 704. 二分查找

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function (nums, target) {
  let left = 0;
  let right = nums.length - 1;
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (nums[mid] > target) {
      right--;
    } else if (nums[mid] < target) {
      left++;
    } else {
      return mid;
    }
  }
  return -1;
};
```

# 35. 搜索插入位置

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var searchInsert = function (nums, target) {
    const n = nums.length;
    let left = 0;
    let right = n - 1;
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (nums[mid] >= target) {
            right--;
        } else if (nums[mid] < target) {
            left++;
        }
    }
    return right + 1;
};
```
