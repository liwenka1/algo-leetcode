# 704. 二分查找

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function (nums, target) {
  let left = 0;
  let right = nums.length - 1;
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (nums[mid] > target) {
      right--;
    } else if (nums[mid] < target) {
      left++;
    } else {
      return mid;
    }
  }
  return -1;
};
```

# 69. x 的平方根 

```js
/**
 * @param {number} x
 * @return {number}
 */
var mySqrt = function (x) {
    if (x < 2) return x;
    let left = 1;
    let right = Math.floor(x / 2);
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (mid * mid === x) return mid;
        if (mid * mid < x) left = mid + 1;
        else right = mid - 1;
    }
    return right;
};
```

# 367. 有效的完全平方数

```js
/**
 * @param {number} num
 * @return {boolean}
 */
var isPerfectSquare = function (num) {
    if (num < 2) return true
    let left = 1;
    let right = Math.floor(num / 2);
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (mid * mid < num) {
            left = mid + 1;
        } else if (mid * mid > num) {
            right = mid - 1;
        } else {
            return true;
        }
    }
    return false;
};
```

# 34. 在排序数组中查找元素的第一个和最后一个位置

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var searchRange = function (nums, target) {
  const n = nums.length;
  const lowerBound = (nums, target) => {
    let left = 0;
    let right = n;
    while (left < right) {
      const mid = Math.floor((left + right) / 2);
      if (nums[mid] < target) {
        left = mid + 1;
      } else {
        right = mid;
      }
    }
    return left;
  };
  const start = lowerBound(nums, target);
  if (start === n || nums[start] !== target) {
    return [-1, -1];
  }
  const end = lowerBound(nums, target + 1);
  return [start, end - 1];
};
```

# 744. 寻找比目标字母大的最小字母

```js
/**
 * @param {string[]} letters
 * @param {string} target
 * @return {string}
 */
var nextGreatestLetter = function (letters, target) {
  const ch = target.charCodeAt(0);
  const n = letters.length;
  let left = 0;
  let right = n - 1;
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (letters[mid].charCodeAt(0) <= ch) {
      left = mid + 1;
    } else if (letters[mid].charCodeAt(0) > ch) {
      right = mid - 1;
    }
  }
  return left < n ? letters[left] : letters[0];
};
```
